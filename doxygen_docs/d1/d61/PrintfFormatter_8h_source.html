250 template<class T>
251 inline char* PrintfFormatter::formatIntDigits(
252  char* bufferEnd,
253  char const* const digits,
254  T value,
255  bool const signedType,
256  typename std::make_unsigned<T>::type const base,
257  int8_t& sign)
258 {
259  if (value != 0U)
260  {
261  if (signedType)
262  {
263  if (static_cast<typename std::make_signed<T>::type>(value) < 0)
264  {
265  value = -value;
266  sign = -1;
267  }
268  else
269  {
270  sign = +1;
271  }
272  }
273  else
274  {
275  sign = +1;
276  }
277 
278  while (value > 0U)
279  {
280  --bufferEnd;
281  *bufferEnd = digits[value % base];
282  value /= base;
283  }
284  }
285  else
286  {
287  sign = 0;
288  --bufferEnd;
289  *bufferEnd = '0';
290  }
291  return bufferEnd;
292 }

295 inline char const* PrintfFormatter::getIntSign(ParamInfo const& paramInfo, int8_t const sign)
296 {
297  char const* pSign = "";
298  if (paramInfo._base == 10U)
299  {
300  switch (paramInfo._datatype)
301  {
302  case ParamDatatype::SINT8:
303  case ParamDatatype::SINT16:
304  case ParamDatatype::SINT32:
305  case ParamDatatype::SINT64:
306  {
307  if (sign < 0)
308  {
309  pSign = "-";
310  }
311  else if ((paramInfo._flags & ParamFlags::FLAG_PLUS) > 0U)
312  {
313  pSign = "+";
314  }
315  else if ((paramInfo._flags & ParamFlags::FLAG_SPACE) > 0U)
316  {
317  pSign = " ";
318  }
319  else
320  {
321  // do nothing
322  }
323  break;
324  }
325  default:
326  {
327  break;
328  }
329  }
330  }
331  return pSign;
332 }

335 inline char const* PrintfFormatter::getIntPrefix(ParamInfo const& paramInfo, int8_t const sign)
336 {
337  char const* pPrefix = "";
338  switch (paramInfo._base)
339  {
340  case 8:
341  {
342  if ((sign > 0) && ((paramInfo._flags & ParamFlags::FLAG_ALT) != 0U))
343  {
344  pPrefix = "0";
345  }
346  break;
347  }
348  case 16:
349  {
350  if ((paramInfo._flags & ParamFlags::FLAG_ALT) != 0U)
351  {
352  pPrefix = ((paramInfo._flags & ParamFlags::FLAG_UPPER) > 0U) ? "0X" : "0x";
353  }
354  break;
355  }
356  default:
357  {
358  break;
359  }
360  }
361  return pPrefix;
362 }
