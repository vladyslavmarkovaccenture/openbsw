181 template<class DataLinkLayer>
182 void DoCanTransportLayerContainer<DataLinkLayer>::shutdown(
183  ShutdownCallbackType const shutdownCallback)
184 {
185  _shutdownPendingCount = 1U;
186  _shutdownCallback = shutdownCallback;
187  for (TransportLayerType& layer : _layers)
188  {
189  {
190  // Even if the value changes between now and the increment, we want to ensure this
191  // function is never about to be incremented when _shutdownPendingCount is ever the max
192  // possible value.
193  estd_assert(
194  _shutdownPendingCount
195  != std::numeric_limits<decltype(_shutdownPendingCount)>::max());
196  ::interrupts::SuspendResumeAllInterruptsScopedLock const lock;
197  ++_shutdownPendingCount;
198  }
199  (void)layer.shutdown(ShutdownDelegate::create<
200  DoCanTransportLayerContainer<DataLinkLayer>,
201  &DoCanTransportLayerContainer<DataLinkLayer>::shutdownDone>(*this));
202  }
203  singleShutdownDone();
204 }

236 template<class DataLinkLayer>
237 void DoCanTransportLayerContainer<DataLinkLayer>::singleShutdownDone()
238 {
239  bool done;
240  {
241  // Even if the value changes between now and the decrement, we want to ensure this function
242  // is never called when _shutdownPendingCount is ever 0.
243  estd_assert(_shutdownPendingCount != 0);
244  ::interrupts::SuspendResumeAllInterruptsScopedLock const lock;
245  --_shutdownPendingCount;
246  done = (_shutdownPendingCount == 0U);
247  }
248  if (done)
249  {
250  _shutdownCallback();
251  }
252 }
