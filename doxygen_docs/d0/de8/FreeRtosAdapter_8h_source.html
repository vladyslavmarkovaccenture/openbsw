293 template<class Binding>
294 void FreeRtosAdapter<Binding>::initTask(TaskInitializer& initializer)
295 {
296  ContextType const context = initializer._context;
297  _stackSizes[static_cast<size_t>(context)] = static_cast<uint32_t>(
298  static_cast<size_t>(initializer._stack.size()) * sizeof(BaseType_t));
299  TaskConfigsType::setTaskConfig(context, initializer._config);
300  if (context == TASK_TIMER)
301  {
302  _timerTaskInitializer = &initializer;
303  _timerTaskName = initializer._name;
304  }
305  else
306  {
307  TaskContextType& taskContext = _taskContexts[static_cast<size_t>(context)];
308  if (context == TASK_IDLE)
309  {
310  _idleTaskInitializer = &initializer;
311  taskContext.initTask(TASK_IDLE, initializer._name, initializer._taskFunction);
312  }
313  else
314  {
315  taskContext.createTask(
316  context,
317  initializer._task,
318  initializer._name,
319  static_cast<UBaseType_t>(context),
320  initializer._stack,
321  initializer._taskFunction);
322  }
323  }
