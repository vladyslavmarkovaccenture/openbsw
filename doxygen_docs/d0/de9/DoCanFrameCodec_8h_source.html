221 template<class DataLinkLayer>
222 CodecResult DoCanFrameCodec<DataLinkLayer>::decodeSingleFrame(
223  ::etl::span<uint8_t const> const& payload,
224  MessageSizeType& messageSize,
225  ::etl::span<uint8_t const>& data) const
226 {
227  size_t const offset = _config._offset;
228  if (checkFrameSize(payload, 2U, _config._singleFrameSize))
229  {
230  messageSize = static_cast<uint8_t>(payload[offset] & 0x0fU);
231  bool const isExtendedSfDlExpected = payload.size() > EXTENDED_SF_DL_EDGE_SIZE;
232  if ((messageSize == 0U) && isExtendedSfDlExpected)
233  {
234  messageSize = payload[offset + 1U];
235  MessageSizeType const checkPayloadSize
236  = messageSize + static_cast<MessageSizeType>(offset) + 2U;
237 
238  if ((checkPayloadSize <= static_cast<MessageSizeType>(_config._singleFrameSize._max))
239  && (static_cast<size_t>(checkPayloadSize) <= payload.size()))
240  {
241  data = ::etl::span<uint8_t const>(
242  &payload[offset + 2U], static_cast<size_t>(messageSize));
243  return CodecResult::OK;
244  }
245  }
246  else if ((messageSize > 0U) && (!isExtendedSfDlExpected))
247  {
248  MessageSizeType const checkPayloadSize
249  = messageSize + static_cast<MessageSizeType>(offset) + 1U;
250  if (checkPayloadSize <= payload.size())
251  {
252  data = ::etl::span<uint8_t const>(
253  &payload[offset + 1U], static_cast<size_t>(messageSize));
254  return CodecResult::OK;
255  }
256  }
257  else
258  {
259  // invalid message size
260  }
261  return CodecResult::INVALID_MESSAGE_SIZE;
262  }
263  return CodecResult::INVALID_FRAME_SIZE;
264 }

266 template<class DataLinkLayer>
267 CodecResult DoCanFrameCodec<DataLinkLayer>::decodeFirstFrame(
268  ::etl::span<uint8_t const> const& payload,
269  MessageSizeType& messageSize,
270  FrameIndexType& frameCount,
271  FrameSizeType& consecutiveFrameDataSize,
272  ::etl::span<uint8_t const>& data) const
273 {
274  if (checkFrameSize(payload, 3U, _config._firstFrameSize))
275  {
276  size_t const offset = _config._offset;
277  messageSize = static_cast<MessageSizeType>(etl::be_uint16_t(&payload[offset]) & 0xFFFU);
278  uint8_t dataStart = 2U;
279  consecutiveFrameDataSize = static_cast<FrameSizeType>(payload.size())
280  - (static_cast<FrameSizeType>(offset) + 1U);
281  if (messageSize == 0U)
282  {
283  if (!checkFrameSize(payload, 6U, _config._firstFrameSize))
284  {
285  return CodecResult::INVALID_FRAME_SIZE;
286  }
287  // Escape sequence, aka message size > 4095 up to uint32_t
288  uint32_t const escapedMessageSize = etl::be_uint32_t(&payload[offset + 2U]);
289 
290  if ((escapedMessageSize <= ESCAPED_SEQ_MESSAGE_SIZE)
291  || (escapedMessageSize > ::etl::numeric_limits<MessageSizeType>::max()))
292  {
293  return CodecResult::INVALID_MESSAGE_SIZE;
294  }
295  messageSize = static_cast<MessageSizeType>(escapedMessageSize);
296  dataStart = 6U;
297  }
298 
299  MessageSizeType const count = (messageSize / consecutiveFrameDataSize);
300  if (count >= etl::numeric_limits<FrameIndexType>::max())
301  {
302  return CodecResult::INVALID_MESSAGE_SIZE;
303  }
304  frameCount = static_cast<FrameIndexType>(count + 1U);
305  bool const msgFitsIntoLongSingleFrame = fitsIntoLongSingleFrame(messageSize);
306  bool const msgFitsIntoShortSingleFrame = fitsIntoShortSingleFrame(messageSize);
307  if ((frameCount > 1U) && (!msgFitsIntoShortSingleFrame) && (!msgFitsIntoLongSingleFrame))
308  {
309  data = ::etl::span<uint8_t const>(
310  &payload[offset + dataStart],
311  static_cast<size_t>(consecutiveFrameDataSize) - dataStart + 1U);
312  return CodecResult::OK;
313  }
314  return CodecResult::INVALID_MESSAGE_SIZE;
315  }
316  return CodecResult::INVALID_FRAME_SIZE;
317 }

363 bool DoCanFrameCodec<DataLinkLayer>::fitsIntoLongSingleFrame(
364  MessageSizeType const messageSize) const

367  estd_assert(_config._singleFrameSize._max >= _config._offset + 2);

372 template<class DataLinkLayer>
373 CodecResult DoCanFrameCodec<DataLinkLayer>::getEncodedFrameCount(
374  MessageSizeType const messageSize,
375  FrameIndexType& frameCount,
376  FrameSizeType& consecutiveFrameDataSize) const
377 {
378  // The max consecutive frame size needs to be at least 2 larger than the configured offset for
379  // this function.
380  estd_assert(_config._consecutiveFrameSize._max >= _config._offset + 2);
381 
382  if (messageSize == 0U)
383  {
384  return CodecResult::INVALID_MESSAGE_SIZE;
385  }
386 
387  bool const msgFitsIntoLongSingleFrame = fitsIntoLongSingleFrame(messageSize);
388  bool const msgFitsIntoShortSingleFrame = fitsIntoShortSingleFrame(messageSize);
389  if (msgFitsIntoShortSingleFrame || msgFitsIntoLongSingleFrame)
390  {
391  consecutiveFrameDataSize = 0U;
392  frameCount = 1U;
393  }
394  else
395  {
396  consecutiveFrameDataSize = _config._consecutiveFrameSize._max - (_config._offset + 1U);
397  MessageSizeType const count = (messageSize / consecutiveFrameDataSize);
398  if (count >= etl::numeric_limits<FrameIndexType>::max())
399  {
400  return CodecResult::INVALID_MESSAGE_SIZE;
401  }
402  frameCount = static_cast<FrameIndexType>(count + 1U);
403  }
404  return CodecResult::OK;
405 }

407 template<class DataLinkLayer>
408 CodecResult DoCanFrameCodec<DataLinkLayer>::encodeDataFrame(
409  ::etl::span<uint8_t>& payload,
410  ::etl::span<uint8_t const> const& data,
411  FrameIndexType const frameIndex,
412  FrameSizeType const consecutiveFrameDataSize,
413  FrameSizeType& consumedDataSize) const
414 {
415  if ((data.size() == 0U) || (data.size() > ::etl::numeric_limits<MessageSizeType>::max()))
416  {
417  return CodecResult::INVALID_FRAME_INDEX;
418  }
419  MessageSizeType const pendingMessageSize = static_cast<MessageSizeType>(data.size());
420 
421  FrameSizeType destOffset;
422  FrameSizeType minFrameSize = 0U;
423  if (frameIndex == 0U)
424  {
425  size_t const offset = _config._offset;
426  if (fitsIntoShortSingleFrame(pendingMessageSize))
427  {
428  payload[offset] = static_cast<uint8_t>(FrameType::SINGLE) << 4U;
429  payload[offset] |= static_cast<uint8_t>(pendingMessageSize) & 0x0FU;
430 
431  destOffset = static_cast<FrameSizeType>(offset) + 1U;
432  consumedDataSize = static_cast<FrameSizeType>(pendingMessageSize);
433  minFrameSize = _config._singleFrameSize._min;
434  }
435  else if (fitsIntoLongSingleFrame(pendingMessageSize))
436  {
437  payload[offset] = static_cast<uint8_t>(FrameType::SINGLE) << 4U;
438  payload[offset + 1U] = static_cast<uint8_t>(pendingMessageSize);
439  destOffset = static_cast<FrameSizeType>(offset) + 2U;
440  consumedDataSize = static_cast<FrameSizeType>(pendingMessageSize);
441  minFrameSize = _config._singleFrameSize._min;
442  }
443  else if (consecutiveFrameDataSize > 0U)
444  {
445  destOffset = static_cast<FrameSizeType>(offset) + 2U;
446  consumedDataSize = consecutiveFrameDataSize - 1U;
447 
448  if (pendingMessageSize <= ESCAPED_SEQ_MESSAGE_SIZE)
449  {
450  etl::be_uint16_ext_t{&payload[offset]} = pendingMessageSize & 0xFFFU;
451  }
452  else
453  {
454  payload[offset] = 0U;
455  payload[offset + 1U] = 0U;
456  etl::be_uint32_ext_t{&payload[offset + 2U]} = pendingMessageSize;
457 
458  destOffset += 4U;
459  consumedDataSize -= 4U;
460  }
461  payload[offset] |= static_cast<uint8_t>(FrameType::FIRST) << 4U;
462  minFrameSize = _config._firstFrameSize._min;
463  }
464  else
465  {
466  return CodecResult::INVALID_FRAME_INDEX;
467  }
468  }
469  else if (consecutiveFrameDataSize > 0U)
470  {
471  uint8_t const offset = _config._offset;
472 
473  payload[offset] = static_cast<uint8_t>(FrameType::CONSECUTIVE) << 4U;
474  payload[offset] |= static_cast<uint8_t>(frameIndex) & 0x0FU;
475 
476  destOffset = offset + 1U;
477  consumedDataSize = static_cast<FrameSizeType>(
478  ::etl::min(static_cast<MessageSizeType>(consecutiveFrameDataSize), pendingMessageSize));
479  minFrameSize = ::etl::max(consumedDataSize, _config._consecutiveFrameSize._min);
480  }
481  else
482  {
483  return CodecResult::INVALID_FRAME_INDEX;
484  }
485 
486  FrameSizeType const payloadSize = (destOffset + consumedDataSize);
487  if (payloadSize > payload.size())
488  {
489  return CodecResult::INVALID_FRAME_SIZE;
490  }
491 
492  (void)::etl::mem_copy(
493  &data[0U],
494  static_cast<size_t>(consumedDataSize),
495  &payload[static_cast<size_t>(destOffset)]);
496  return adjustFrame(payload, payloadSize, minFrameSize);
497 }

499 template<class DataLinkLayer>
500 CodecResult DoCanFrameCodec<DataLinkLayer>::encodeFlowControlFrame(
501  ::etl::span<uint8_t>& payload,
502  FlowStatus const flowStatus,
503  uint8_t const blockSize,
504  uint8_t const encodedMinSeparationTime) const
505 {
506  size_t const offset = _config._offset;
507  if (payload.size() < (offset + 3U))
508  {
509  return CodecResult::INVALID_FRAME_SIZE;
510  }
511 
512  payload[offset] = static_cast<uint8_t>(FrameType::FLOW_CONTROL) << 4U;

521 template<class DataLinkLayer>
522 CodecResult DoCanFrameCodec<DataLinkLayer>::adjustFrame(
523  ::etl::span<uint8_t>& payload,
524  FrameSizeType const payloadSize,
525  FrameSizeType const minFrameSize) const
526 {
527  FrameSizeType paddedFrameSize = ::etl::max(payloadSize, minFrameSize);
528  if (!_mapper.mapFrameSize(paddedFrameSize))
529  {
530  return CodecResult::INVALID_FRAME_SIZE;
531  }
532 
533  if (paddedFrameSize > payloadSize)
534  {
535  if (paddedFrameSize > payload.size())
536  {
537  return CodecResult::INVALID_FRAME_SIZE;
538  }
539 
540  ::etl::mem_set(
541  &payload[static_cast<size_t>(payloadSize)],
542  static_cast<size_t>(paddedFrameSize) - payloadSize,
543  _config._filler);
544  }
545  payload = payload.first(static_cast<size_t>(paddedFrameSize));
546  return CodecResult::OK;
547 }
