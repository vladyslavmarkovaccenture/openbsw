212 inline uint32_t DoCanParameters::decodeMinSeparationTime(uint8_t const encodedMinSeparationTime)
213 {
214  if (encodedMinSeparationTime <= 0x7FU)
215  {
216  return static_cast<uint32_t>(encodedMinSeparationTime) * 1000U;
217  }
218 
219  if ((encodedMinSeparationTime >= 0xF1U) && (encodedMinSeparationTime <= 0xF9U))
220  {
221  return (static_cast<uint32_t>(encodedMinSeparationTime) - 0xF0U) * 100U;
222  }
223 
224  return 0x7fU * 1000U;
225 }
226 
227 inline uint8_t DoCanParameters::encodeMinSeparationTime(uint32_t const minSeparationTimeUs)
228 {
229  // Encode 100-900us STmin values
230  if ((minSeparationTimeUs >= 100U) && (minSeparationTimeUs < 1000U))
231  {
232  return static_cast<uint8_t>(minSeparationTimeUs / 100U) + 0xF0U;
233  }
234 
235  // Encode millisecond STmin values
236  if (minSeparationTimeUs < (0x7FU * 1000U))
237  {
238  return static_cast<uint8_t>(minSeparationTimeUs / 1000U);
239  }
240 
241  // Default to max valid STmin for all other cases
242  return 0x7FU;
243 }
