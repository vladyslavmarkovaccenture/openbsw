250 template<class Adapter>
251 template<typename T>
252 void TaskInitializer<Adapter>::create(
253  ContextType const context,
254  char const* const name,
255  TaskObjectType& task,
256  T& stack,
257  TaskFunctionType const taskFunction,
258  TaskConfigType const& config)
259 {
260  ::etl::span<uint8_t> bytes = ::etl::make_span(stack).template reinterpret_as<uint8_t>();
261  align(alignof(StackType_t), bytes);
262  StackSliceType const stackSlice = bytes.template reinterpret_as<StackType_t>();
263  estd_assert((stackSlice.size() * sizeof(StackType_t)) >= sizeof(TaskInitializer));
264  new (stackSlice.data()) TaskInitializer(context, name, task, stackSlice, taskFunction, config);
265 }
