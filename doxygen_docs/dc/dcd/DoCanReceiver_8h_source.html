200 template<class DataLinkLayer>
201 void DoCanReceiver<DataLinkLayer>::init()
202 {
203  // Perform assert checks in init() so we can't possibly assert during construction, which could
204  // cause serious problems for some of our projects (HW getting locked due to cyclic resets)
205  // Ensure the block pool block size >= MessageReceiverType size, otherwise
206  // _maxFirstFrameDataSize will be an invalid garbage value
207  estd_assert(_messageReceiverPool.max_item_size() >= sizeof(MessageReceiverType));
208  // Ensure the following subtraction is less than FrameSizeType's max value, otherwise
209  // _maxFirstFrameDataSize will be an invalid garbage value
210  estd_assert(
211  _messageReceiverPool.max_item_size() - sizeof(MessageReceiverType)
212  < std::numeric_limits<FrameSizeType>::max());
213 }

228 template<class DataLinkLayer>
229 void DoCanReceiver<DataLinkLayer>::firstDataFrameReceived(
230  ConnectionType const& connection,
231  MessageSizeType const messageSize,
232  FrameIndexType const frameCount,
233  FrameSizeType const consecutiveFrameDataSize,
234  ::etl::span<uint8_t const> const& data)
235 {
236  DataLinkAddressPairType const& dataLinkAddressPair = connection.getDataLinkAddressPair();
237  // functional routing entries are defined with dest TA as INVALID
238  // only functional single frames (frameCount == 1) are allowed
239  if ((frameCount > 1U)
240  && (dataLinkAddressPair.getTransmissionAddress() == DataLinkLayerType::INVALID_ADDRESS))
241  {
242  char formatBuffer[FORMAT_BUFFER_SIZE];
243  ::util::logger::Logger::warn(
244  _loggerComponent,
245  "DoCanReceiver(%s)::firstDataFrameReceived(%s): functional first frames (RA: "
246  "0x%x) are not allowed!",
247  getName(),
248  _addressConverter.formatDataLinkAddress(
249  connection.getDataLinkAddressPair().getReceptionAddress(), formatBuffer),
250  dataLinkAddressPair.getReceptionAddress());
251  }
252  else if (!_messageReceiverPool.full())
253  {
254  RemoveGuard const guard(this);
255  // Check data size to avoid memory corruption
256  if (data.size() <= _maxFirstFrameDataSize)
257  {
258  MessageReceiverType* messageReceiver;
259  {
260  ::interrupts::SuspendResumeAllInterruptsScopedLock const lock;
261  uint8_t* receiver = reinterpret_cast<uint8_t*>(
262  _messageReceiverPool.template allocate<MessageReceiverType>());
263 
264  auto const firstFrameCopy = copyFirstFrameData(receiver, data);
265  bool const blocked
266  = handlePendingMessageReceivers(dataLinkAddressPair.getReceptionAddress());
267  messageReceiver = ::new (receiver) MessageReceiverType(
268  connection,
269  messageSize,
270  frameCount,
271  consecutiveFrameDataSize,
272  _parameters.getMaxBlockSize(),
273  _parameters.getEncodedMinSeparationTime(),
274  firstFrameCopy,
275  blocked);
276  _messageReceivers.push_back(*messageReceiver);
277  }
278  handleTransitions(
279  *messageReceiver, handleTransition(*messageReceiver), "firstDataFrameReceived");
280  }
281  else
282  {
283  char formatBuffer[FORMAT_BUFFER_SIZE];
284  ::util::logger::Logger::error(
285  _loggerComponent,
286  "DoCanReceiver(%s)::firstDataFrameReceived(%s): Received %d bytes, "
287  "expected max %d bytes!",
288  getName(),
289  _addressConverter.formatDataLinkAddress(
290  connection.getDataLinkAddressPair().getReceptionAddress(), formatBuffer),
291  data.size(),
292  _maxFirstFrameDataSize);
293  }
294  }
295  else
296  {
297  char formatBuffer[FORMAT_BUFFER_SIZE];
298  ::util::logger::Logger::warn(
299  _loggerComponent,
300  "DoCanReceiver(%s)::firstDataFrameReceived(%s): No empty message receiver "
301  "found!",
302  getName(),
303  _addressConverter.formatDataLinkAddress(
304  connection.getDataLinkAddressPair().getReceptionAddress(), formatBuffer));
305  }
306 }

308 template<class DataLinkLayer>
309 void DoCanReceiver<DataLinkLayer>::consecutiveDataFrameReceived(
310  DataLinkAddressType const receptionAddress,
311  uint8_t const sequenceNumber,
312  ::etl::span<uint8_t const> const& data)
313 {
314  MessageReceiverType* const messageReceiver = findMessageReceiver(receptionAddress);
315  if ((messageReceiver == nullptr) || (!messageReceiver->isConsecutiveFrameExpected()))
316  {
317  char formatBuffer[FORMAT_BUFFER_SIZE];
318  ::util::logger::Logger::warn(
319  _loggerComponent,
320  "DoCanReceiver(%s)::consecutiveDataFrameReceived(%s): unexpected consecutive frame!",
321  getName(),
322  _addressConverter.formatDataLinkAddress(receptionAddress, formatBuffer));
323  return;
324  }
325 
326  FrameSizeType const expectedSize = messageReceiver->getExpectedConsecutiveFrameDataSize();
327  if (data.size() < expectedSize)
328  {
329  char formatBuffer[FORMAT_BUFFER_SIZE];
330  ::util::logger::Logger::warn(
331  _loggerComponent,
332  "DoCanReceiver(%s)::consecutiveDataFrameReceived(%s): consecutive frame length "
333  "(%d) less than expected (%d)!",
334  getName(),
335  _addressConverter.formatDataLinkAddress(receptionAddress, formatBuffer),
336  data.size(),
337  expectedSize);
338  return;
339  }
340 
341  RemoveGuard const guard(this);
342  handleTransitions(
343  *messageReceiver,
344  messageReceiver->consecutiveFrameReceived(sequenceNumber, expectedSize, data),
345  "consecutiveDataFrameReceived");
346  return;
347 }

349 template<class DataLinkLayer>
350 void DoCanReceiver<DataLinkLayer>::cyclicTask(uint32_t const nowUs)
351 {
352  {
353  RemoveGuard const guard(this);
354  for (typename MessageReceiverListType::iterator it = _messageReceivers.begin();
355  it != _messageReceivers.end();
356  ++it)
357  {
358  if (!it->updateTimer(nowUs))
359  {
360  break;
361  }
362  ::interrupts::SuspendResumeAllInterruptsScopedLock const lock;
363  handleTransitions(*it, it->expired(), "cyclicTask");
364  }
365  }
366  // Sort after removal.
367  if (_timersUpdated)
368  {
369  ::interrupts::SuspendResumeAllInterruptsScopedLock const lock;
370  _messageReceivers.sort();
371  _timersUpdated = false;
372  }
373 }

383 template<class DataLinkLayer>
384 void DoCanReceiver<DataLinkLayer>::processMessageReceivers()
385 {
386  RemoveGuard const guard(this);
387  for (typename MessageReceiverListType::iterator it = _messageReceivers.begin();
388  it != _messageReceivers.end();
389  ++it)
390  {
391  MessageReceiverType& messageReceiver = *it;
392  ::interrupts::SuspendResumeAllInterruptsScopedLock const lock;
393  if (messageReceiver.isAllocating())
394  {
395  handleTransitions(
396  messageReceiver,
397  allocateTransportMessage(messageReceiver, true),
398  "processMessageReceivers");
399  if (messageReceiver.getMessage() == nullptr)
400  {
401  break;
402  }
403  }
404  }

418 template<class DataLinkLayer>
419 void DoCanReceiver<DataLinkLayer>::handleResult(
420  MessageReceiverType& messageReceiver,
421  ReceiveResult const result,
422  char const* const functionName)
423 {
424  if (result.hasTransition())
425  {
426  resetTimer(messageReceiver);
427  if (messageReceiver.getState() != ReceiveState::DONE)
428  {
429  _timersUpdated = true;
430  }
431  }
432  if (result.getMessage() != ReceiveMessage::NONE)
433  {
434  char stringBuffer[FORMAT_BUFFER_SIZE];
435  auto const formattedAddress = _addressConverter.formatDataLinkAddress(
436  messageReceiver.getReceptionAddress(), stringBuffer);
437  // create a log for each separately to allow compiler to skip them if no logs are
438  // defined
439  switch (result.getMessage())
440  {
441  case ReceiveMessage::ILLEGAL_STATE:
442  {
443  ::util::logger::Logger::warn(
444  _loggerComponent,
445  "DoCanReceiver(%s)::%s(%s): Illegal state 0x%x!",
446  getName(),
447  functionName,
448  formattedAddress,
449  result.getParam());
450  break;
451  }
452  case ReceiveMessage::ALLOCATION_RETRY_COUNT_EXCEEDED:
453  {
454  ::util::logger::Logger::warn(
455  _loggerComponent,
456  "DoCanReceiver(%s)::%s(%s): Allocation retry count exceeded",
457  getName(),
458  functionName,
459  formattedAddress);
460  break;
461  }
462  case ReceiveMessage::RX_TIMEOUT_EXPIRED:
463  {
464  ::util::logger::Logger::warn(
465  _loggerComponent,
466  "DoCanReceiver(%s)::%s(%s): Rx timeout",
467  getName(),
468  functionName,
469  formattedAddress);
470  break;
471  }
472  case ReceiveMessage::BAD_SEQUENCE_NUMBER:
473  {
474  ::util::logger::Logger::warn(
475  _loggerComponent,
476  "DoCanReceiver(%s)::%s(%s): Frame with bad sequence number "
477  "received",
478  getName(),
479  functionName,
480  formattedAddress);
481  break;
482  }
483  case ReceiveMessage::PROCESSING_FAILED:
484  {
485  ::util::logger::Logger::warn(
486  _loggerComponent,
487  "DoCanReceiver(%s)::%s(%s): Processing failed",
488  getName(),
489  functionName,
490  formattedAddress);
491  break;
492  }
493  default:
494  {
495  break;
496  }
497  }
498  }
499 }

501 template<class DataLinkLayer>
502 ReceiveResult DoCanReceiver<DataLinkLayer>::handleTransition(MessageReceiverType& messageReceiver)
503 {
504  switch (messageReceiver.getState())
505  {
506  case ReceiveState::ALLOCATE:
507  {
508  return allocateTransportMessage(messageReceiver, false);
509  }
510  case ReceiveState::SEND:
511  {
512  return sendFlowControlFrame(messageReceiver);
513  }
514  case ReceiveState::PROCESSING:
515  {
516  return startProcessingTransportMessage(messageReceiver);
517  }
518  case ReceiveState::DONE:
519  {
520  return release(messageReceiver);
521  }
522  default:
523  {
524  break;
525  }
526  }
527  return ReceiveResult(false);
528 }

530 template<class DataLinkLayer>
531 ReceiveResult DoCanReceiver<DataLinkLayer>::allocateTransportMessage(
532  MessageReceiverType& messageReceiver, bool const peek)
533 {
534  DoCanTransportAddressPair const transportAddressPair
535  = messageReceiver.getTransportAddressPair();
536  ::transport::TransportMessage* message = nullptr;
537  if (!messageReceiver.isBlocked())
538  {
539  ::transport::ITransportMessageProvider::ErrorCode const result
540  = _messageProvidingListener.getTransportMessage(
541  _busId,
542  transportAddressPair.getSourceId(),
543  transportAddressPair.getTargetId(),
544  messageReceiver.getMessageSize(),
545  messageReceiver.getFirstFrameData(),
546  message);
547  if ((result != ::transport::ITransportMessageProvider::ErrorCode::TPMSG_OK)
548  && (result
549  != ::transport::ITransportMessageProvider::ErrorCode::TPMSG_NO_MSG_AVAILABLE))
550  {
551  char stringBuffer[FORMAT_BUFFER_SIZE];
552  auto const formattedAddress = _addressConverter.formatDataLinkAddress(
553  messageReceiver.getReceptionAddress(), stringBuffer);
554  switch (result)
555  {
556  case ::transport::ITransportMessageProvider::ErrorCode::TPMSG_INVALID_SRC_ID:
557  {
558  ::util::logger::Logger::warn(
559  _loggerComponent,
560  "DoCanReceiver(%s)::allocateTransportMessage(%s): Illegal "
561  "source "
562  "id.",
563  getName(),
564  formattedAddress);
565  break;
566  }
567  case ::transport::ITransportMessageProvider::ErrorCode::TPMSG_INVALID_TGT_ID:
568  {
569  ::util::logger::Logger::warn(
570  _loggerComponent,
571  "DoCanReceiver(%s)::allocateTransportMessage(%s): Illegal "
572  "target "
573  "id.",
574  getName(),
575  formattedAddress);
576  break;
577  }
578  default:
579  {
580  ::util::logger::Logger::warn(
581  _loggerComponent,
582  "DoCanReceiver(%s)::allocateTransportMessage(%s): "
583  "No buffer available (error: 0x%x). Message discarded.",
584  getName(),
585  formattedAddress,
586  result);
587  break;
588  }
589  }
590  return messageReceiver.cancel();
591  }
592 
593  if (message != nullptr)
594  {
595  message->resetValidBytes();
596  message->setSourceId(transportAddressPair.getSourceId());
597  message->setTargetId(transportAddressPair.getTargetId());
598  message->setPayloadLength(messageReceiver.getMessageSize());
599  }
600  // Don't do anything in this scope if message == nullptr
601  }
602  if ((message == nullptr) && peek)
603  {
604  return ReceiveResult(false);
605  }
606 
607  return messageReceiver.allocated(message, _parameters.getMaxAllocateRetryCount());
608 }

651 template<class DataLinkLayer>
652 ReceiveResult DoCanReceiver<DataLinkLayer>::release(MessageReceiverType& messageReceiver)
653 {
654  DataLinkAddressType const receptionAddress = messageReceiver.getReceptionAddress();
655  ::transport::TransportMessage* const message = messageReceiver.release();
656  if (message != nullptr)
657  {
658  _messageProvidingListener.releaseTransportMessage(*message);
659  }
660  for (typename MessageReceiverListType::iterator it = _messageReceivers.begin();
661  it != _messageReceivers.end();
662  ++it)
663  {
664  if (it->isBlocked() && (it->getReceptionAddress() == receptionAddress))
665  {
666  it->setBlocked(false);
667  break;
668  }
669  }
670  // Ensure we don't wrap _releasedReceiverCount back to 0
671  estd_assert(
672  _releasedReceiverCount != std::numeric_limits<decltype(_releasedReceiverCount)>::max());
673  ++_releasedReceiverCount;
674  return ReceiveResult(false);
675 }

677 template<class DataLinkLayer>
678 void DoCanReceiver<DataLinkLayer>::resetTimer(MessageReceiverType& messageReceiver) const
679 {
680  auto const nowUs = _parameters.nowUs();
681  switch (messageReceiver.getTimeout())
682  {
683  case ReceiveTimeout::ALLOCATE:
684  {
685  messageReceiver.setTimer(
686  nowUs + (static_cast<uint32_t>(_parameters.getWaitAllocateTimeout()) * 1000U));
687  break;
688  }
689  case ReceiveTimeout::RX:
690  {
691  messageReceiver.setTimer(
692  nowUs + (static_cast<uint32_t>(_parameters.getWaitRxTimeout()) * 1000U));
693  break;
694  }
695  default:
696  {
697  messageReceiver.setTimer(nowUs);
698  break;
699  }
700  }

703 template<class DataLinkLayer>
704 bool DoCanReceiver<DataLinkLayer>::handlePendingMessageReceivers(
705  DataLinkAddressType const receptionAddress)
706 {
707  bool blocked = false;
708  for (typename MessageReceiverListType::iterator it = _messageReceivers.begin();
709  it != _messageReceivers.end();
710  ++it)
711  {
712  if (it->getReceptionAddress() == receptionAddress)
713  {
714  if (it->getFrameCount() > 1U)
715  {
716  char formatBuffer[FORMAT_BUFFER_SIZE];
717 
718  ::util::logger::Logger::info(
719  _loggerComponent,
720  "DoCanReceiver(%s)::handlingPendingMessageReceivers(%s): "
721  "Segmented transfer cancelled due to new first frame.",
722  getName(),

724  it->getReceptionAddress(), formatBuffer));
725  handleTransitions(*it, it->cancel(), "handlingPendingMessageReceivers");
726  }
727  else
728  {
729  blocked = true;
730  }
731  }
732  }
733  return blocked;
734 }

761 template<class DataLinkLayer>
762 void DoCanReceiver<DataLinkLayer>::releaseRemoveLock()
763 {
764  // Don't allow _removeLockCount to wrap around
765  estd_assert(_removeLockCount != 0);
766  ::interrupts::SuspendResumeAllInterruptsScopedLock const lock;
767  --_removeLockCount;
768  if ((_removeLockCount == 0U) && (_releasedReceiverCount > 0U))
769  {
770  typename MessageReceiverListType::iterator it = _messageReceivers.begin();
