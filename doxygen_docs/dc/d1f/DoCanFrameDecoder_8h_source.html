44 CodecResult DoCanFrameDecoder<Codec>::decodeFrame(
45  DoCanConnection<DataLinkLayerType> const& connection,
46  ::etl::span<uint8_t const> const& payload,
47  IDoCanFrameReceiver<DataLinkLayerType>& receiver)
48 {
49  CodecType const& codec = connection.getFrameCodec();
50  FrameType frameType;
51  CodecResult result = codec.decodeFrameType(payload, frameType);
52  if (result == CodecResult::OK)
53  {
54  switch (frameType)
55  {
56  case FrameType::SINGLE:
57  {
58  ::etl::span<uint8_t const> data;
59  MessageSizeType messageSize;
60  result = codec.decodeSingleFrame(payload, messageSize, data);
61  if (result == CodecResult::OK)
62  {
63  receiver.firstDataFrameReceived(connection, messageSize, 1U, 0U, data);
64  }
65  break;
66  }
67  case FrameType::FIRST:
68  {
69  ::etl::span<uint8_t const> data;
70  MessageSizeType messageSize;
71  FrameIndexType frameCount;
72  FrameSizeType consecutiveDataFrameSize;
73  result = codec.decodeFirstFrame(
74  payload, messageSize, frameCount, consecutiveDataFrameSize, data);
75  if (result == CodecResult::OK)
76  {
77  receiver.firstDataFrameReceived(
78  connection, messageSize, frameCount, consecutiveDataFrameSize, data);
79  }
80  break;
81  }
82  case FrameType::CONSECUTIVE:
83  {
84  ::etl::span<uint8_t const> data;
85  uint8_t sequenceNumber;
86  result = codec.decodeConsecutiveFrame(payload, sequenceNumber, data);
87  if (result == CodecResult::OK)
88  {
89  receiver.consecutiveDataFrameReceived(
90  connection.getDataLinkAddressPair().getReceptionAddress(),
91  sequenceNumber,
92  data);
93  }
94  break;
95  }
96  case FrameType::FLOW_CONTROL:
97  {
98  FlowStatus flowStatus;
99  uint8_t blockSize;
100  uint8_t encodedMinSeparationTime;
101  result = codec.decodeFlowControlFrame(
102  payload, flowStatus, blockSize, encodedMinSeparationTime);
103  if (result == CodecResult::OK)
104  {
105  receiver.flowControlFrameReceived(
106  connection.getDataLinkAddressPair().getReceptionAddress(),
107  flowStatus,
108  blockSize,
109  encodedMinSeparationTime);
110  }
111  break;
112  }
113  default:
114  {
115  return CodecResult::INVALID_FRAME_TYPE;
116  }
117  }
118  }
119  return result;
120 }
