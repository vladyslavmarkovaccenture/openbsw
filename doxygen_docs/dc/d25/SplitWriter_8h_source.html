128 template<size_t N>
129 void SplitWriter<N>::commit()
130 {
131  if (_bufNum == INVALID_BUF)
132  {
133  return;
134  }
135  for (size_t i = 0; i < N; i++)
136  {
137  if (i == _bufNum)
138  {
139  // skip the one used as buffer to do it last
140  continue;
141  }
142  auto const dest = _destinations[i];
143  auto const tmp = dest->allocate(_buffer.size());
144  if (tmp.size() < _buffer.size())
145  {
146  drops[i] += 1;
147  continue;
148  }
149  (void)::etl::mem_copy(_buffer.begin(), _buffer.size(), tmp.begin());
150  sent[i] += 1;
151  dest->commit();
152  }
153 
154  sent[_bufNum] += 1;
155  _destinations[_bufNum]->commit();
156  _bufNum = INVALID_BUF;
157 }
