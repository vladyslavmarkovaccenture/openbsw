61  uint32_t size() const
62  {
63  // Volatile is needed to to make sure the read is not optimized away when new elements have
64  // been added to the queue by the other core.
65  uint32_t const txPos = static_cast<uint32_t const volatile&>(_sent);
66  return (txPos >= _received) ? (txPos - _received) : (txPos + (2U * _maxSize)) - _received;
67  }

94  void takeSnapshot()
95  {
96  uint32_t const currentSize = size();
97  if (0U != currentSize)
98  {
99  _stats.loadSnapshot += static_cast<uint16_t>(currentSize);
100  ++_stats.processingCounter;
101  }
102  _stats.realLoadSnapshot += static_cast<uint16_t>(currentSize);
103  ++_stats.realProcessingCounter;
104  if (_stats.previousSnapshot == 0U)
105  {
106  _stats.maxFillRate = static_cast<uint8_t>(currentSize);
107  _stats.previousSnapshot = static_cast<uint8_t>(currentSize);
108  }
109  else
110  {
111  if (currentSize > _stats.previousSnapshot)
112  {
113  auto const diff = (currentSize - _stats.previousSnapshot);
114  if (diff > _stats.maxFillRate)
115  {
116  _stats.maxFillRate = static_cast<uint8_t>(diff);
117  }
118  }
119  _stats.previousSnapshot = static_cast<uint8_t>(currentSize);
120  }
