35 {
36  value = ((value >> 1U) & 0x55U) | (static_cast<uint8_t>(value << 1U) & 0xAAU);
37  value = ((value >> 2U) & 0x33U) | (static_cast<uint8_t>(value << 2U) & 0xCCU);
38  return (value >> 4U) | static_cast<uint8_t>(value << 4U);
