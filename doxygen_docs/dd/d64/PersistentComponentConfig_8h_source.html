42 {
43  ::etl::span<uint8_t const> const readBuffer = _persistenceManager.readMapping(_buffer);
44  if (readBuffer.size() > 1U)
45  {
46  uint8_t componentCount = static_cast<uint8_t>(readBuffer.size()) - 1U;
47  if (componentCount > IndexUpperBound)
48  {
49  componentCount = IndexUpperBound;
50  }
51  uint8_t const crc = readBuffer[0U];
52  if (crc == getComponentCrc(componentCount))
53  {
54  for (uint8_t idx = 0U; idx < componentCount; ++idx)
55  {
56  this->setLevel(
57  idx,

59  }
60  }
