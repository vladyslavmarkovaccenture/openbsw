294 template<class Binding>
295 inline EventMaskType TaskContext<Binding>::waitEvents()
296 {
297  EventMaskType eventMask = 0U;
298  uint32_t ticks = Binding::WAIT_EVENTS_TICK_COUNT;
299  uint32_t nextDelta;
300  bool const hasDelta = _timer.getNextDelta(getSystemTimeUs32Bit(), nextDelta);
301  if (hasDelta)
302  {
303  ticks = static_cast<uint32_t>((nextDelta + (Config::TICK_IN_US - 1U)) / Config::TICK_IN_US);
304  }
305  if (xTaskNotifyWait(0U, WAIT_EVENT_MASK, &eventMask, ticks) != 0)
306  {
307  return eventMask;
308  }
309  else if (hasDelta)
310  {
311  return TimerEventPolicyType::EVENT_MASK;
312  }
313  else
314  {
315  return 0U;
316  }
317 }
