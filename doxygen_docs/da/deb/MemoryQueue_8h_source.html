64  static size_t advanceIndex(size_t index, size_t const size)
65  {
66  index = (index + size + sizeof(SIZE_TYPE)) % (2 * CAPACITY);
67 
68  // Check if enough contiguous space is available at the end of the array and wrap
69  // otherwise skipping the too small piece of memory.
70  size_t const space = (((2 * CAPACITY) - index) % CAPACITY);
71  if (space < (MAX_ELEMENT_SIZE + sizeof(SIZE_TYPE)))
72  {
73  index = (index + space) % (2 * CAPACITY);
74  }
75  return index;
76  }

231 template<size_t CAPACITY, size_t MAX_ELEMENT_SIZE, typename SIZE_TYPE>
232 ::etl::span<uint8_t>
233 MemoryQueue<CAPACITY, MAX_ELEMENT_SIZE, SIZE_TYPE>::Writer::allocate(size_t const size) const
234 {
235  // Set allocated zero prevent a subsequent call to commit() to have
236  // effects in case this allocation fails. That is important if this is a
237  // reallocation, i.e. a previous call to allocate() succeeded.
238  _txData.allocated = 0U;
239  if ((size > MAX_ELEMENT_SIZE) || (size == 0))
240  {
241  return {};
242  }
243  auto const freeBytes = available();
244  _txData.minAvailable = ::etl::min(freeBytes, _txData.minAvailable);
245  if (freeBytes == 0)
246  {
247  return {};
248  }
249  size_t const index = _txData.sent.load() % CAPACITY;
250  _txData.allocated = size;
251  return ::etl::span<uint8_t>(&_txData.data[index + sizeof(SIZE_TYPE)], size);
252 }

254 template<size_t CAPACITY, size_t MAX_ELEMENT_SIZE, typename SIZE_TYPE>
255 void MemoryQueue<CAPACITY, MAX_ELEMENT_SIZE, SIZE_TYPE>::Writer::commit()
256 {
257  // Prevent accidentally committing random data if allocate has not been called or
258  // previous allocation was unsuccessful.
259  if (_txData.allocated == 0U)
260  {
261  return;
262  }
263  size_t writeIndex = _txData.sent.load();
264  size_t const index = writeIndex % CAPACITY;
265  SIZE_TYPE const size = static_cast<SIZE_TYPE>(_txData.allocated);
266  ::etl::unaligned_type_ext<SIZE_TYPE, etl::endian::big>{&_txData.data[index]}
267  = static_cast<SIZE_TYPE>(size);
268 
269  writeIndex = advanceIndex(writeIndex, size);
270  _txData.allocated = 0U;
271  // Store writeIndex last to ensure data consistency.
272  _txData.sent.store(writeIndex);
273 }

368 template<size_t CAPACITY, size_t MAX_ELEMENT_SIZE, typename SIZE_TYPE>
369 size_t
370 MemoryQueue<CAPACITY, MAX_ELEMENT_SIZE, SIZE_TYPE>::TxData::available(RxData const& rxData) const
371 {
372  size_t const writeIndex = sent.load();
373  size_t const readIndex = rxData.received.load();
374 
375  size_t usedBytes;
376  if (writeIndex < readIndex)
377  {
378  usedBytes = (writeIndex + (2 * CAPACITY)) - readIndex;
379  }
380  else
381  {
382  usedBytes = writeIndex - readIndex;
383  }
384 
385  size_t freeBytes = CAPACITY - usedBytes;
386  if (freeBytes < (MAX_ELEMENT_SIZE + sizeof(SIZE_TYPE)))
387  {
388  freeBytes = 0;
389  }
390  return freeBytes;
391 }
